Lottie.js 官網 

            <script>
                // 加載 Lottie 動畫
                const animationContainer = document.getElementById('lottie');
            
                const animation = lottie.loadAnimation({
                    container: animationContainer, // 動畫容器
                    path: 'animation.json', // 確保這個路徑正確，並且能訪問到 animation.json 文件
                    renderer: 'svg', // 渲染器選項（'svg'、'canvas' 或 'html'）
                    loop: true, // 是否循環播放
                    autoplay: true, // 是否自動播放
                });
            </script>
