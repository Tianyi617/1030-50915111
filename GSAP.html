GSAP 動畫示例

GSAP官網 

            <script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.11.5/gsap.min.js"></script>
            <script>
                const box = document.getElementById("box");
                let tween = gsap.to(box, { x: 400, duration: 2, repeat: -1, yoyo: true, paused: true });
            
                document.getElementById("start").onclick = () => tween.play();
                document.getElementById("pause").onclick = () => tween.pause();
                document.getElementById("resume").onclick = () => tween.resume();
                document.getElementById("reverse").onclick = () => tween.timeScale(-1);
                document.getElementById("restart").onclick = () => tween.restart();
            </script>
