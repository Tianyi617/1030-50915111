Chart.js 柱狀圖示例

Chart.js 官網 

            <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
            <script>
                // 獲取畫布元素
                const ctx = document.getElementById('myChart').getContext('2d');
            
                // 創建柱狀圖
                const myChart = new Chart(ctx, {
                    type: 'bar', // 圖表類型
                    data: {
                        labels: ['紅色', '藍色', '黃色', '綠色', '紫色', '橙色'], // X 軸標籤
                        datasets: [{
                            label: '數量',
                            data: [12, 19, 3, 5, 2, 3], // 對應的數據
                            backgroundColor: [
                                'rgba(255, 99, 132, 0.2)',
                                'rgba(54, 162, 235, 0.2)',
                                'rgba(255, 206, 86, 0.2)',
                                'rgba(75, 192, 192, 0.2)',
                                'rgba(153, 102, 255, 0.2)',
                                'rgba(255, 159, 64, 0.2)'
                            ],
                            borderColor: [
                                'rgba(255, 99, 132, 1)',
                                'rgba(54, 162, 235, 1)',
                                'rgba(255, 206, 86, 1)',
                                'rgba(75, 192, 192, 1)',
                                'rgba(153, 102, 255, 1)',
                                'rgba(255, 159, 64, 1)'
                            ],
                            borderWidth: 1 // 邊框寬度
                        }]
                    },
                    options: {
                        scales: {
                            y: {
                                beginAtZero: true // Y 軸從 0 開始
                            }
                        }
                    }
                });
            </script>
