AOS動畫

AOS官網 

        <!-- AOS JS -->
        <script src="https://cdn.jsdelivr.net/npm/aos@2.3.4/dist/aos.js"></script>
        <script>
            // 延遲初始化 AOS
            window.addEventListener('load', function () {
                AOS.init({
                    duration: 800, // 動畫時間
                    offset: 100    // 動畫偏移量
                });
            });
        
            const container = document.getElementById('container');
            const colors = ['blanchedalmond', 'lightcoral', 'lightblue', 'lightgreen', 'lightpink', 'lightyellow', 'lavender', 'peachpuff', 'lightgray', 'lightcyan'];
            const animations = ['fade-up', 'fade-down', 'fade-right', 'fade-left', 'zoom-in', 'zoom-out', 'flip-left', 'flip-right', 'slide-down', 'slide-up'];
        
            for (let i = 0; i < 10; i++) {
                const newDiv = document.createElement('div');
                newDiv.setAttribute('data-aos', animations[i % animations.length]);
                newDiv.setAttribute('data-aos-delay', i * 100);
        
                newDiv.style.margin = '40px auto';
                newDiv.style.backgroundColor = colors[i % colors.length]; // 使用不同顏色
                newDiv.style.padding = '20px';
                newDiv.style.height = '150px'; // 設定區塊高度
                newDiv.style.width = '80%'; // 設定區塊寬度
                newDiv.style.display = 'flex'; // 使用 flexbox 進行布局
                newDiv.style.justifyContent = 'center'; // 水平置中
                newDiv.style.alignItems = 'center'; // 垂直置中
                newDiv.style.borderRadius = '10px'; // 圓角
                newDiv.textContent = animations[i % animations.length]; // 將文字內容改為對應的效果名字
                newDiv.style.boxShadow = '0 0 10px rgba(0, 0, 0, 0.1)'; // 陰影
        
                container.appendChild(newDiv);
            }
        </script>
