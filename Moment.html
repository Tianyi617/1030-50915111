moment.js時區示例

momentjs官網 

        <script>
            function updateTimes() {
                // 獲取當前時間
                const now = moment();
        
                // 倫敦時間
                const londonTime = now.tz("Europe/London").format('YYYY-MM-DD HH:mm:ss');
                document.getElementById("london").innerText = `倫敦: ${londonTime}`;
        
                // 台灣時間
                const taiwanTime = now.tz("Asia/Taipei").format('YYYY-MM-DD HH:mm:ss');
                document.getElementById("taiwan").innerText = `台灣: ${taiwanTime}`;
        
                // 美國東部時間
                const usaTime = now.tz("America/New_York").format('YYYY-MM-DD HH:mm:ss');
                document.getElementById("usa").innerText = `美國東部: ${usaTime}`;
            }
        
            // 每秒更新一次時間
            setInterval(updateTimes, 1000);
            updateTimes(); // 初始顯示時間
        
        </script>
